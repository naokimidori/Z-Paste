   - duration: 0.2 秒
   - timingFunction: easeInEaseOut
   - 方向: 从屏幕底部滑入/滑出

2. **位置计算**:
```swift
private func calculatePositions() -> (onScreen: CGFloat, offScreen: CGFloat) {
    guard let screen = NSScreen.main else { return (0, 0) }
    let screenFrame = screen.visibleFrame
    let onScreenY = screenFrame.origin.y  // 屏幕底部
    let offScreenY = screenFrame.origin.y - windowHeight - 10  // 屏幕外下方
    return (onScreenY, offScreenY)
}
```

3. **showWindow 实现**:
```swift
func showWindow() {
    guard let panel = panel, !isAnimating else { return }
    isAnimating = true

    NSApp.activate(ignoringOtherApps: true)
    panel.makeKeyAndOrderFront(nil)

    let positions = calculatePositions()
    NSAnimationContext.runAnimationGroup { context in
        context.duration = 0.2
        context.timingFunction = CAMediaTimingFunction(name: .easeInEaseOut)
        panel.animator().setFrameOrigin(
            NSPoint(x: calculateXPosition(), y: positions.onScreen)
        )
    } completionHandler: { [weak self] in
        self?.isAnimating = false
    }
}
```

4. **hideWindow 实现**:
```swift
func hideWindow() {
    guard let panel = panel, panel.isVisible, !isAnimating else { return }
    isAnimating = true

    let positions = calculatePositions()
    NSAnimationContext.runAnimationGroup { context in
        context.duration = 0.2
        context.timingFunction = CAMediaTimingFunction(name: .easeInEaseOut)
        panel.animator().setFrameOrigin(
            NSPoint(x: calculateXPosition(), y: positions.offScreen)
        )
    } completionHandler: { [weak self] in
        panel.orderOut(nil)
        self?.isAnimating = false
    }
}
```

5. **toggleWindow 实现**:
```swift
func toggleWindow() {
    if panel?.isVisible == true {
        hideWindow()
    } else {
        showWindow()
    }
}
```

6. **多显示器支持**:
   - 获取鼠标所在显示器: `NSEvent.mouseLocation`
   - 找到包含鼠标位置的 screen: `NSScreen.screens.first { $0.frame.contains(mouseLocation) }`
